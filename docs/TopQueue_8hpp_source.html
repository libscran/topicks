98 void push_internal(std::pair<Stat_, Index_> gene, const Compare_ cmp, const Skip_ skip) {
99 if (my_bound.has_value()) {
100 if (skip(*my_bound, gene.first)) {
101 return;
102 } else if (my_open_bound && *my_bound == gene.first) {
103 return;
104 }
105 }
106
107 if (sanisizer::is_less_than(my_queue.size(), my_top)) {
108 my_queue.push_back(std::move(gene));
109 std::push_heap(my_queue.begin(), my_queue.end(), cmp);
110 return;
111 }
112
113 if (my_top == 0) {
114 return;
115 }
116 const auto& top = my_queue.front();
117 if (skip(top.first, gene.first)) {
118 return;
119 }
120
121 if (!my_keep_ties) {
122 if (top.first != gene.first || gene.second < top.second) {
123 std::pop_heap(my_queue.begin(), my_queue.end(), cmp);
124 my_queue.back() = std::move(gene);
125 std::push_heap(my_queue.begin(), my_queue.end(), cmp);
126 }
127 return;
128 }
129
130 if (top.first == gene.first) {
131 // Always making sure that all elements in my_ties are of equal or lower priority than the tied element in my_queue.
132 if (gene.second >= top.second) {
133 my_ties.push_back(std::move(gene));
134 std::push_heap(my_ties.begin(), my_ties.end(), SortTies());
135 } else {
136 auto retop = top;
137 std::pop_heap(my_queue.begin(), my_queue.end(), cmp);
138 my_queue.back() = std::move(gene);
139 std::push_heap(my_queue.begin(), my_queue.end(), cmp);
140 my_ties.push_back(std::move(retop));
141 std::push_heap(my_ties.begin(), my_ties.end(), SortTies());
142 }
143 return;
144 }
145
146 if (my_top == 1) {
147 my_queue.front() = std::move(gene);
148 return;
149 }
150
151 // The idea is to check whether the top 2 elements of the queue are tied,
152 // and if so, shift the top value into the tie queue.
153 auto retop = top;
154 std::pop_heap(my_queue.begin(), my_queue.end(), cmp);
155 if (retop.first == my_queue.front().first) {
156 my_ties.push_back(std::move(retop));
157 std::push_heap(my_ties.begin(), my_ties.end(), SortTies());
158 } else {
159 my_ties.clear();
160 }
161 my_queue.back() = std::move(gene);
162 std::push_heap(my_queue.begin(), my_queue.end(), cmp);
163 }
